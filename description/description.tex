    class Server(Thread):
      def run():
        server_socket = create_server_socket()
        while True:
          client_socket = server_socket.accept()
          data = client_socket.read_all(timeout=1)
          if data:
            print(client_socket.address)
            client_socket.send(data)
          client_socket.close()
